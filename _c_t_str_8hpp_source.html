97 // 0001 is the most basic; for every character it either creates an std::integer_sequence containing the one character
98 // or an empty std::integer_sequence if the character was out of bounds.
99 // It looks kinda funky though. Here's how it works:
100 //
101 // make_helper takes 3 arguments, the type of the string, the character at this point in the string, and whether the length of the string is too long for CTSTR_MAX_STRING_SIZE
102 // The first two are obvious. The reason the third is there (and repeated for every character) is because there's just nowhere else to put it...
103 #define string_to_type_0001(str, n) \
104 typename ::ctstr::detail::make_helper \
105 < \
106  std::decay_t<decltype(*str)>, \
107  ::ctstr::detail::c_str_lit_at_or_eol(str, CTSTR_STRLEN(str), 0x00 + n), \
108  CTSTR_STRLEN(str) < CTSTR_MAX_STRING_SIZE \
109 >::type
